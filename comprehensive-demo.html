Usage Examples

How to integrate Shadow Toolkit into your projects

📄Plain HTML

<script src="shadow-cursor-engine.js"></script>
<script>
  shadowCursor.setTheme('minecraft');
</script>

<shadow-2d-physics style="width:300px;height:300px;">
  <video src="asset.webm" autoplay loop muted></video>
</shadow-2d-physics>

<shadow-holo-card>
  <img src="product.png">
</shadow-holo-card>

📦NPM Package

# Install
npm install @shadow-toolkit/core

# Import
import { ShadowCursorEngine } from '@shadow-toolkit/core';

// Use
const cursor = new ShadowCursorEngine();
cursor.setTheme('tech');

⚛️Framework Integration

// React
function CursorProvider() {
  useEffect(() => {
    const cursor = new ShadowCursorEngine();
    cursor.setTheme('gaming');
    return () => cursor.destroy();
  }, []);
  
  return (
    <div>
      <shadow-2d-physics>
        <img src="character.png" />
      </shadow-2d-physics>
    </div>
  );
}

// Vue 3
<template>
  <shadow-2d-physics ref="physics">
    <img src="asset.webm" />
  </shadow-2d-physics>
</template>

<script setup>
import { onMounted, onUnmounted, ref } from 'vue';
import { ShadowCursorEngine } from '@shadow-toolkit/core';

const physics = ref(null);
let cursor = null;

onMounted(() => {
  cursor = new ShadowCursorEngine();
  cursor.setTheme('retro');
});

onUnmounted(() => {
  if (cursor) cursor.destroy();
});
</script>
